Step 1: Organize Your Conversations

First, create a dedicated folder:

mkdir conversations

Now open your script (local_ai.py) and add this near the top:

import json
import os

CONVERSATIONS_DIR = "conversations"
os.makedirs(CONVERSATIONS_DIR, exist_ok=True)

os.makedirs(..., exist_ok=True) ensures the folder exists — quietly creating it if needed, doing nothing if it’s already there.

Clean. Safe. No drama.

This is where all your chats will live — each as a separate, human-readable file.

# ai_chat_issue8.py #issue 8 Ctrl+Click
from openai import OpenAI
import json # new import
import os # new import


# ---- New section for Conversations Directory -------------------
CONVERSATIONS_DIR = "conversations"
os.makedirs(CONVERSATIONS_DIR, exist_ok=True)


# ---- Conversations Directory Helper Fucntions ------------------
def list_conversations():
"""Print all available conversation files"""
try:
files = [f for f in os.listdir(CONVERSATIONS_DIR) if f.endswith(".json")]
if not files:
print("📭 No saved conversations yet.")
else:
print("\n📁 Saved conversations:")
for i, f in enumerate(files, 1):
name = f[:-5] # Remove '.json'
print(f" {i}. {name}")
return files
except Exception as e:
print(f"Error reading conversations: {e}")
return []


def choose_conversation():
files = list_conversations()

print("\n🔁 What would you like to do?")
print("1. Start a new conversation")
print("2. Load an existing one")

choice = input("> ").strip()

if choice == "1":
name = input("Name for this chat: ").strip()
if not name:
name = "untitled"
return f"{name}.json", []

elif choice == "2" and files:
try:
num = int(input("Choose a number: ")) - 1
if 0 <= num < len(files):
filename = files[num]
path = os.path.join(CONVERSATIONS_DIR, filename)
with open(path, 'r', encoding='utf-8') as f:
history = json.load(f)
print(f"💬 Loaded '{filename[:-5]}'.")
return filename, history
else:
print("Invalid number. Starting fresh.")
name = input("Name for new chat: ") or "untitled"
return f"{name}.json", []
except ValueError:
print("Not a number. Starting fresh.")
name = input("Name for new chat: ") or "untitled"
return f"{name}.json", []
else:
print("No saved chats. Starting new one.")
name = input("Name for this chat: ") or "untitled"
return f"{name}.json", []


# ---- Select or Create Conversations ------------------
filename, conversation_history = choose_conversation()
filepath = os.path.join(CONVERSATIONS_DIR, filename)




# Connect to your local AI
client = OpenAI(
base_url="http://localhost:8080/v1",
api_key="not-needed"
)


# starting main loop
while True:
user_input = input("You: ")
if user_input.lower() == "!quit":
print("Bye! Staying local, staying free.")
break


# Add user message to history
conversation_history.append({"role": "user", "content": user_input})


# Send to local AI
response = client.chat.completions.create(
model="local-model",
messages=conversation_history,
max_tokens=500,
temperature=0.35,
)


# Extract and print reply
ai_reply = response.choices[0].message.content
print(f"AI: {ai_reply}")


# Save AI response to history
conversation_history.append({"role": "assistant", "content": ai_reply})


# Save updated history
with open(filepath, 'w', encoding='utf-8') as f:
json.dump(conversation_history, f, indent=2, ensure_ascii=False)

🛠️ Troubleshooting Tips

❌ “File not found” when loading
Make sure the filename includes .json.
Check that CONVERSATIONS_DIR exists and has the right name.

❌ Conversation resets after closing
Did you forget to save at the end of the loop?
Make sure the with open(...) as f: block runs after each reply.

❌ Strange characters in the file
Always use encoding='utf-8' when opening files.
Avoid special characters in filenames — stick to letters, numbers, and hyphens.
